function siHei() {
    let navn = document.getElementById("navnInput").value;

        alert("Du må skrive inn et navn.");
    } else {
        alert("Hallo " + navn + "! Jeg ønsker deg en fin dag!");
    }
}
</script>

<input type="text" id="navnInput">
<button onclick="siHei()">Send</button>
